        <#if vars?is_sequence><#return /></#if>

<#if vars[k]?size &gt; 0><#t>

                        </#if><#t>
                    </#list>                    "                    </#if>
                <#else>
<#if vars[k]?is_number || vars[k]?is_boolean ><#local temp=vars[k]?c /><#t>
                    <#else><#local temp = vars[k]?string /><#t>
                    </#if>                    <#if temp != "">

                </#if>

        </#list>

</#macro>
